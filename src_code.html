JAVASCRIPT CODE:

<script async defer>
const addRowButton = document.getElementById("button_tableAddRow");
const addColumnButton = document.getElementById("button_tableAddColumn");

const table = document.getElementById("table");

let rowCount = 0;
let columnCount = 0;

let newRow;
let node;

AddRow();


function AddRow() {

     if (rowCount < 30) {
     rowCount++;
     console.log("Added row " + rowCount);

     newRow = document.createElement("tr");
     node = document.createElement("input");
     NewCellHandle();
     table.appendChild(newRow);

     newColumn = document.createElement("td");
     node = document.createElement("input");
     NewCellHandle();
     newColumn.appendChild(node);
     table.appendChild(newColumn);

     } else {
         console.log("Cannot add row! 30 is the maximum!");
         addRowButton.innerHTML = "Max reached";

     }

}

function AddColumn() {

   if (columnCount < 50) {
      columnCount++;
      console.log("Added column " + columnCount);

      newColumn = document.createElement("td");
      node = document.createElement("input");
      NewCellHandle();

      newColumn.appendChild(node);
      table.appendChild(newColumn);

   } else {
      console.log("Cannot add column! 50 is the maximum!");
      addColumnButton.innerHTML = "Max reached";

   }

}
function ReloadTable() {
     location.reload();
}

function NewCellHandle() {
   node.className = "newCellInput";

   node.style.width = "100%";
   console.log(node.className);

   table.style.borderCollapse = "collapse";
}

// Called when page is loaded (opacity effect)
function PageLoadEffect() {
   document.body.style.opacity = 0;
   const bodyElem = document.getElementById("body");
   let opac = 0;

   let intervalNum = 2;
   let inter = setInterval(frame, intervalNum);

   function frame() {
      if (opac == 100) {
         clearInterval(inter);
      } else {
         document.body.style.opacity = opac / 100;
         opac++;
      }

   }
}
</script>
